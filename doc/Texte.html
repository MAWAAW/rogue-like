def loadFile()
        @tabTexte = Array.new;
        fic = File.open("langue/"+@langue+".txt",'r')
        fic.each_line{ |ligne|
                tab = ligne.split(";")
                @tabTexte.push(tab)
        }
        #self.debugAfficherTab
        return self;
end
