def donnePoidsTotal()
        poidsTotal=0
        @listObjet.each { |i|
                element = i
                poids_element = element.getPoids() # récupère le poids de l'élément
                poidsTotal = poidsTotal+poids_element
        }
        return poidsTotal
end

def verifierPoidsTotal(unObjet)
        poidsTotal = donnePoidsTotal()
        if( poidsTotal+unObjet.getPoids() <= self.getPoidsMax() )
                return true
        else
                return false
        end
end
