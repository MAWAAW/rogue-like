# File carte/Case.rb, line 30
def voisine( d )
        case d
                when Direction::REPOS
                        return self
                when Direction::HAUT
                        return @carte.getCase( @coord.coordX, @coord.coordY-1 )
                when Direction::DROITE
                        return @carte.getCase( @coord.coordX+1, @coord.coordY )
                when Direction::BAS
                        return @carte.getCase( @coord.coordX, @coord.coordY+1 )
                when Direction::GAUCHE
                        return @carte.getCase( @coord.coordX-1, @coord.coordY )
        end
end
